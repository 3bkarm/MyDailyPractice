## [Container With Most Water](https://leetcode.com/problems/container-with-most-water/description/?envType=daily-question&envId=2025-10-04)
```java
class Solution {
    public int maxArea(int[] height) {
        int n = height.length;
        int[] prefixMax = new int[n];
        int[] suffixMax = new int[n];
        for (int i = 0; i < n; ++i) {
            prefixMax[i] = height[i];
            suffixMax[i] = height[i];
        }
        for (int i = 1; i < n; ++i) {
            prefixMax[i] = Math.max(prefixMax[i], prefixMax[i - 1]);
        }
        for (int i = n - 2; i >= 0; --i) {
            suffixMax[i] = Math.max(suffixMax[i], suffixMax[i + 1]);
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int lo = i + 1, hi = n - 1, to = i;
            while (lo <= hi) {
                int mid = lo + (hi - lo) / 2;
                if (suffixMax[mid] >= height[i]) {
                    lo = mid + 1;
                    to = mid;
                } else {
                    hi = mid - 1;
                }
            }
            if (i < to) {
                ans = Math.max(ans, (to - i) * height[i]);
            }
            
            lo = 0;
            hi = i - 1;
            to = i;
            while (lo <= hi) {
                int mid = lo + (hi - lo) / 2;
                if (prefixMax[mid] >= height[i]) {
                    hi = mid - 1;
                    to = mid;
                } else {
                    lo = mid + 1;
                }
            }
            
            if (i > to) {
                ans = Math.max(ans, (i - to) * height[i]);
            }
        }
        return ans;
    }
}
```
