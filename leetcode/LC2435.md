## [Paths in Matrix Whose Sum Is Divisible by K](https://leetcode.com/problems/paths-in-matrix-whose-sum-is-divisible-by-k?envType=daily-question&envId=2025-11-26)
```java
class Solution {
    private final static int mod = 1000_000_007;
    public int numberOfPaths(int[][] grid, int k) {
        int n = grid.length;
        int m = grid[0].length;
        int[][][] dp = new int[n][m][k];
        dp[n - 1][m - 1][0] = 1;
        for (int i = n - 1; i >= 0; --i) {
            for (int j = m - 1; j >= 0; --j) {
                if (i == n - 1 && j == m - 1) continue;
                for (int remainder = 0; remainder < k; ++remainder) {
                    int ans = 0;
                    if (i + 1 < n) {
                        int c = (remainder + grid[i + 1][j]) % k;
                        ans += dp[i + 1][j][c];
                        if (ans >= mod) ans -= mod;
                    }
                    if (j + 1 < m) {
                        int c = (remainder + grid[i][j + 1]) % k;
                        ans += dp[i][j + 1][c];
                        if (ans >= mod) ans -= mod;
                    }
                    dp[i][j][remainder] = ans;
                }
            }
        }
        return dp[0][0][ grid[0][0] % k ];
    }
}
```
