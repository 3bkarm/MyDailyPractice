## [Maximize the Minimum Powered City](https://leetcode.com/problems/maximize-the-minimum-powered-city?envType=daily-question&envId=2025-11-07)

```java
class Solution {
    private boolean can(long target, int d, int k, long[] count) {
        int n = count.length;
        long over = 0;
        long[] add = new long[n];
        for (int i = 0; i < n; ++i) {
            int l = Math.max(0, i - d);
            int r = Math.min(n - 1, i + d);
            long have = count[r] + over;
            if (l > 0) {
                have -= count[l - 1];
            }
            long need = Math.max(0l, target - have);
            if (need > k) {
                return false;
            }
            k -= need;
            over += need;
            add[r] += need;
            if (i - l == d) {
                over -= add[l];
            }
        }
        return true;
    }
    public long maxPower(int[] stations, int r, int k) {
        int n = stations.length;
        long[] count = new long[n];
        for (int i = 0; i < n; ++i) {
            count[i] = stations[i];
            if (i > 0) {
                count[i] += count[i - 1];
            }
        }
        long lo = 1, hi = 1000_000_000_000l, ans = 0;
        while (lo <= hi) {
            long mid = lo + (hi - lo) / 2;
            if ( can(mid, r, k, count) ) {
                lo = mid + 1;
                ans = mid;
            } else {
                hi = mid - 1;
            }
        }
        return ans;
    }
}
```
