## [Largest Triangle Area](https://leetcode.com/problems/largest-triangle-area/description/?envType=daily-question&envId=2025-09-27)
```java
class Solution {
    public double largestTriangleArea(int[][] points) {
        int n = points.length, big = 0;
        for (int m = 0; m < n; ++m) {
            for (int x = 0; x < n; ++x) {
                if (x == m) {
                    continue;
                }
                for (int y = 0; y < n; ++y) {
                    if (y == m || y == x) {
                        continue;
                    }
                    int ux = points[x][0] - points[m][0];
                    int vx = points[x][1] - points[m][1];
                    
                    int uy = points[y][0] - points[m][0];
                    int vy = points[y][1] - points[m][1];
                    
                    int c = (ux * vy) - (uy * vx);
                    if (c < 0) {
                        c = -c;
                    }
                    
                    big = Math.max(big, c);
                }
            }
        }
        return 1.0 * big / 2.0;
    }
}
```
