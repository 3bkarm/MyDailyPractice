## [Count the Number of Substrings With Dominant Ones](https://leetcode.com/problems/count-the-number-of-substrings-with-dominant-ones?envType=daily-question&envId=2025-11-15)
``` java
class Solution {
    public int numberOfSubstrings(String s) {
        int n = s.length();
        int[] nxt = new int[n];
        for (int i = n - 1, to = n; i >= 0; --i) {
            nxt[i] = to;
            if (s.charAt(i) == '0') {
                to = i;
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (s.charAt(i) == '1') {
                ans += (nxt[i] - i);
            }
        }
        for (int b = 1; b + b * b <= n; ++b) {
            int l = 0, r = 0, z = 0, o = 0;
            while (l < n) {
                while (r < n && z < b) {
                    if (s.charAt(r) == '0') {
                        ++z;
                    } else {
                        ++o;
                    }
                    ++r;
                }
                if (z != b) {
                    break;
                }
                while (r < n && o < b * b) {
                    if (s.charAt(r) == '1') {
                        ++o;
                    } else {
                        break;
                    }
                    ++r;
                }
                if (o >= b * b) {
                    ans += nxt[r - 1] - (r - 1);
                }
                if (s.charAt(l) == '0') {
                    --z;
                } else {
                    --o;
                }
                ++l;
            }
        }
        return ans;
    }
}
```
