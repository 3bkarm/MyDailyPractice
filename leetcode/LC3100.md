## [Water Bottles II](https://leetcode.com/problems/water-bottles-ii/description/?envType=daily-question&envId=2025-10-02)
```java
class Solution {
    private int sum(int n) {
        return n * (n + 1) / 2;
    }
    public int maxBottlesDrunk(int numBottles, int numExchange) {
        int empty = 0, ans = 0;
        while (numBottles > 0 || empty >= numExchange) {
            empty += numBottles;
            ans += numBottles;
            int lo = 1, hi = 10000, get = 0;
            while (lo <= hi) {
                int mid = lo + (hi - lo) / 2;
                if ( sum(numExchange + mid - 1) - sum(numExchange - 1) <= empty ) {
                    lo = mid + 1;
                    get = mid;
                } else {
                    hi = mid - 1;
                }
            }
            numBottles = get;
            empty -= ( sum(numExchange + get - 1) - sum(numExchange - 1) );
            numExchange += get;
        }
        return ans;
    }
}
```
