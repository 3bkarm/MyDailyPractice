## [Swim in Rising Water](https://leetcode.com/problems/swim-in-rising-water/description/?envType=daily-question&envId=2025-10-06)
```java
class Solution {
    private int n;
    private int[][] time;
    PriorityQueue<int[]> pq;

    private static int[] dx = new int[]{ 0, 0, -1, 1 };
    private static int[] dy = new int[]{ -1, 1, 0, 0 };

    private void initial(int n) {
        this.n = n;
        time = new int[n][n];
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                time[i][j] = Integer.MAX_VALUE;
            }
        }
        pq = new PriorityQueue<>( (a, b) -> a[0] - b[0] );
        time[0][0] = 0;
        pq.add(new int[]{ 0, 0, 0 });
    }

    public int swimInWater(int[][] grid) {
        initial(grid.length);
        while ( !pq.isEmpty() ) {
            int[] t = pq.poll();
            int x = t[1], y = t[2];
            for (int move = 0, tox, toy; move < 4; ++move) {
                tox = x + dx[move];
                toy = y + dy[move];
                if (tox >= 0 && tox < n && toy >= 0 && toy < n &&
                        t[0] + Math.max( 0, Math.max(grid[x][y], grid[tox][toy]) - t[0] ) < time[tox][toy]) {
                    time[tox][toy] = t[0] + Math.max( 0, Math.max(grid[x][y], grid[tox][toy]) - t[0] );
                    pq.add(new int[]{ time[tox][toy], tox, toy });
                }
            }
        }
        return time[n - 1][n - 1];
    }
}
```
