## [Set Intersection Size At Least Two](https://leetcode.com/problems/set-intersection-size-at-least-two?envType=daily-question&envId=2025-11-20)
```java
class Solution {
    public int intersectionSizeTwo(int[][] intervals) {
        int n = intervals.length;
        Arrays.sort(intervals, (a, b) -> {
            if (a[1] == b[1]) {
                return b[0] - a[0];
            }
            return a[1] - b[1];
        });

        int ans = 2, a = intervals[0][1] - 1, b = a + 1;
        for (int i = 1; i < n; ++i) {
            int l = intervals[i][0];
            int r = intervals[i][1];
            if (l <= a) {
                continue;
            }
            if (l > b) {
                ans += 2;
                a = r - 1;
                b = r;
            } else {
                ++ans;
                a = b;
                b = r;
            }
        }

        return ans;
    }
}
```
