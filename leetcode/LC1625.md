## [Lexicographically Smallest String After Applying Operations](https://leetcode.com/problems/lexicographically-smallest-string-after-applying-operations/description/?envType=daily-question&envId=2025-10-19)
```java
class Solution {
    private int gcd(int a, int b) {
        if (a == 0) {
            return b;
        }
        return gcd(b % a, a);
    }
    public String findLexSmallestString(String s, int a, int b) {
        int n = s.length();

        char[] ans = new char[n];
        for (int i = 0; i < n; ++i) {
            ans[i] = '9';
        }

        int g = gcd(n, b);
        for (int start = 0; start < n; start += g) {
            for (int odd = 0; odd < 10; ++odd) {
                for (int even = 0; even <= (b % 2) * 9; ++even) {
                    char[] compare = new char[n];
                    for (int at = 0, i = start; at < n; ++at, i = (i + 1) % n) {
                        int d = s.charAt(i) - '0';
                        if (i % 2 == 0) {
                            d = (d + even * a) % 10;
                        } else {
                            d = (d + odd * a) % 10;
                        }
                        compare[at] = (char)(d + '0');
                    }
                    boolean check = true;
                    for (int i = 0; i < n; ++i) {
                        if (ans[i] == compare[i]) {
                            continue;
                        }
                        if (ans[i] < compare[i]) {
                            check = false;
                        }
                        break;
                    }
                    if (check) {
                        for (int i = 0; i < n; ++i) {
                            ans[i] = compare[i];
                        }
                    }
                }
            }
        }
        return new String(ans);
    }
}
```
